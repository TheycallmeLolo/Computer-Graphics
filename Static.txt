void reshape(int w ,int h ){
	float ratio;
	if( h == 0 )
		h = 1;
	ratio = (float)w / h ;
	glViewport(0,0,w,h);
	glMatrixMode(GL_PROJECTION);
	glLoadIdentity();
	if( w >= h )
		gluOrtho2D(-10*ratio,10*ratio,-10,10);
	else
		gluOrtho2D(-10,10,-10/ratio,10/ratio);
	glMatrixMode(GL_MODELVIEW);
}
